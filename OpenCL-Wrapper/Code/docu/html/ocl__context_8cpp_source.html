51 ocl::Context::Context(cl_context id, bool shared) :
52  _id(id), _programs(), _queues(), _events(), _memories(), _samplers(), _devices(), _activeQueue(NULL), _activeProgram(NULL)
53 {
54  TRUE_ASSERT(_id != 0, "Context not valid");
55 
56 
57  size_t num_devices = 0;
58  OPENCL_SAFE_CALL( clGetContextInfo (_id, CL_CONTEXT_DEVICES, 0, NULL, &num_devices) );
59  if(num_devices>0){
60  cl_device_id* devices = new cl_device_id[num_devices];
61  OPENCL_SAFE_CALL( clGetContextInfo (_id, CL_CONTEXT_DEVICES, num_devices, devices, NULL) );
62  for(size_t i = 0; i < num_devices; ++i) this->_devices.push_back(ocl::Device(devices[i]));
63  delete[] devices;
64  this->create(shared);
65  }

195  cl_int status;
196  std::vector<cl_device_id> dev = this->cl_devices();
197  if(shared == false)
198  {
199  _id = clCreateContext(NULL, dev.size(), dev.data(), NULL, NULL, &status);
200  OPENCL_SAFE_CALL(status);
201  TRUE_ASSERT(_id != 0, "Could not create Context");
202  return;
203  }
204 
205 // __comment("Creating a shared context");
206 
207  // TODO: check if shared context possible.
208 
209  #ifdef __OPENGL__

211  std::vector<cl_context_properties> properties;
212  #ifdef __APPLE__
213  CGLContextObj glContext = CGLGetCurrentContext();
214  CGLShareGroupObj shareGroup = CGLGetShareGroup(glContext);
215  properties.push_back(CL_CONTEXT_PROPERTY_USE_CGL_SHAREGROUP_APPLE);
216  properties.push_back((cl_context_properties)shareGroup);
217  properties.push_back(0);

221  properties.push_back(CL_GLX_DISPLAY_KHR);

223  properties.push_back(CL_CONTEXT_PLATFORM);
224  properties.push_back((cl_context_properties) this->devices().front().platform());
225  properties.push_back(0);

257 void ocl::Context::release()
258 {
259  if(this->_id == 0) return;
260 
261  for(auto it = _programs.begin(); it != _programs.end();){
262  ocl::Program *p = *it; ++it;
263  this->release(p);
264  }
265  for(auto it = _queues.begin(); it != _queues.end();){
266  ocl::Queue *q = *it; ++it;
267  this->release(q);
268  }
269  for(auto it = _events.begin(); it != _events.end();){
270  ocl::Event *e = *it; ++it;
271  this->release(e);
272  }
273  for(auto it = _memories.begin(); it != _memories.end();){
274  ocl::Memory *m = *it; ++it;
275  this->release(m);
276  }
277  for(auto it = _samplers.begin(); it != _samplers.end();){
278  ocl::Sampler *m = *it; ++it;
279  this->release(m);
280  }
281 
282  OPENCL_SAFE_CALL( clReleaseContext( _id ) );
283 
284  this->_activeProgram = 0;
285  this->_activeQueue = 0;
286  this->_id = 0;
287 }
