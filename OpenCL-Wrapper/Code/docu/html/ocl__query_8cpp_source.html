57 vector<cl_platform_id> ocl::platforms()
58 {
59  cl_uint numPlatforms = 0;
60 
61  // Query for the number of recongnized platforms
62  OPENCL_SAFE_CALL( clGetPlatformIDs(0, NULL, &numPlatforms));
63  TRUE_ASSERT(numPlatforms > 0, "No Platforms detected.");
64 
65  vector<cl_platform_id> pltfs(numPlatforms, 0);
66  OPENCL_SAFE_CALL( clGetPlatformIDs(numPlatforms, pltfs.data(), NULL));
67 
68  return pltfs;

72 vector<cl_device_id> ocl::devices(cl_platform_id id)
73 {
74  TRUE_ASSERT(id != 0, "Invalid Platform");
75  cl_uint numDevices = 0;
76  // Retrieve the number of devices present on the specified platform of the specified deviceType
77  OPENCL_SAFE_CALL( clGetDeviceIDs(id, CL_DEVICE_TYPE_ALL, 0, NULL, &numDevices) );
78  TRUE_ASSERT(numDevices > 0, "No devices detected.");
79 
80  vector<cl_device_id> devs(numDevices,0);
81  OPENCL_SAFE_CALL( clGetDeviceIDs(id, CL_DEVICE_TYPE_ALL, numDevices, &devs[0], NULL) );
82  return devs;
83 }

267 double ocl::execTime(cl_event event)
268 {
269  cl_ulong start, end;
270 
271  OPENCL_SAFE_CALL( clGetEventProfilingInfo(event, CL_PROFILING_COMMAND_END, sizeof(cl_ulong), &end, NULL) );
272  OPENCL_SAFE_CALL( clGetEventProfilingInfo(event, CL_PROFILING_COMMAND_START, sizeof(cl_ulong), &start, NULL) );
273 
274  return double(end - start); // returns in nanoseconds
275 }
