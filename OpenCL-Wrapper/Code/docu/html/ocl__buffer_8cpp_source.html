126 void ocl::Buffer::create(size_t size_bytes, Access access )
127 {
128  TRUE_ASSERT(this->_context != 0, "Context not valid - cannot create buffer");
129 
130  TRUE_ASSERT(this->id() == nullptr, "Cannot create buffer twice. Please release buffer.");
131 
132  cl_mem_flags flags = access;
133 
134  if(this->context()->devices().size() == 1 &&
135  this->context()->devices().at(0).type() == ocl::device_type::CPU){
136  flags |= ocl::Buffer::AllocHost;
137  }
138 
139  cl_int status;
140  _id = clCreateBuffer(this->_context->id(), flags, size_bytes, NULL, &status);
141  OPENCL_SAFE_CALL( status );
142  TRUE_ASSERT(_id != 0, "could not create buffer");
143 }
