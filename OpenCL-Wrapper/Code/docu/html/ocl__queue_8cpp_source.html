109 void ocl::Queue::create(ocl::Context * ctxt)
110 {
111  if(ctxt == 0){
112  TRUE_ASSERT(_context != 0, "Must have a context.");
113  }
114  else {
115  TRUE_ASSERT(ctxt == _context || _context == NULL, "Cannot have different contexts for the same program.");
116  _context = ctxt;
117  }
118 
119  cl_int status;
120 
121 #if CL_VERSION_2_0
122  if ( supportsAtLeast2Point0( device().platform() ) )
123  {
124  cl_queue_properties propties[] = {
125  CL_QUEUE_PROPERTIES, this->properties(),
126  0
127  };
128 
129  _id = clCreateCommandQueueWithProperties( this->context().id(), this->device().id(), propties, &status );
130  }
131  else
132 #endif
133  {
134  _id = clCreateCommandQueue(this->context().id(), this->device().id(), this->properties(), &status);
135  }
136 
137  OPENCL_SAFE_CALL(status);
138  TRUE_ASSERT(_id != NULL, "clCreateCommandQueue failed\n");
139  _context->insert(this);
140 }
