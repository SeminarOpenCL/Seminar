73  is.open (name.c_str(), mode);

193 inline void printND(const T* A, const utl::Dim &dims, std::vector<int> &offsets, int &size, const utl::Storage & s, std::ostream &out)
194 {
195  out << "cat("<< size+1 <<",..." << std::endl;
196  for(int d = 0; d < dims.at(size); ++d){
197  int oldsize = size;
198  if(size >= 2){
199  int dimt = 1;
200  for(int t = 0; t < size; ++t)
201  dimt *= dims.at(t);
202  offsets.at(size-2) = d*dimt + offsets.at(size-1);
203  }
204  if(size == 2) print2D(A, dims, offsets[0], s, out);
205  else if(size > 2) printND(A, dims, offsets, --size, s, out);
206  size = oldsize;
207  if(d < dims.at(size)-1) out << ",";
208  out << "..." << std::endl;
209  }
210  out << ")";
