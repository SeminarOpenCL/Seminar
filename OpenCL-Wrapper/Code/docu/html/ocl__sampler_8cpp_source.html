72 void ocl::Sampler::create(AdressingMode amode, FilterMode fmode, bool normalized)
73 {
74  TRUE_ASSERT(this->_context != 0, "Context not valid - cannot create Sampler");
75 
76  cl_bool coords = normalized ? CL_TRUE : CL_FALSE;
77  cl_addressing_mode cl_amode = amode;
78  cl_filter_mode cl_fmode = fmode;
79  cl_int status;
80 
81 // #ifdef CL_VERSION_2_0
82 #ifdef OPENCL_V2_0
83  cl_sampler_properties props[] = {
84  CL_SAMPLER_NORMALIZED_COORDS, coords,
85  CL_SAMPLER_ADDRESSING_MODE, cl_amode,
86  CL_SAMPLER_FILTER_MODE, cl_fmode,
87  0
88  };
89 
90  this->_id = clCreateSamplerWithProperties( this->_context->id(), props, &status );
91 #else
92  this->_id = clCreateSampler(this->_context->id(), coords, cl_amode, cl_fmode, &status);
93 #endif
94 
95  OPENCL_SAFE_CALL(status);
96  TRUE_ASSERT(this->_id != 0, "Could not create Sampler.");
97 }
