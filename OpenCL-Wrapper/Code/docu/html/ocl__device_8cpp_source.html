348 static bool supportsExtension( std::string const& extensionsString, char const* extension )
349 {
350  auto const len = std::strlen( extension );
351  auto p = extensionsString.c_str();
352 
353  while ( *p != '\0' )
354  {
355  auto n = std::strcspn( p, " " );
356 
357  if ( len == n && 0 == strncmp( p, extension, n ) ) return true;
358  p += ++n;
359  }
360 
361  return false;
362 }
