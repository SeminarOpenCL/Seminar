15 void utl::ProfilePass::run()
16 {
17  //size_t j = 0;
18 
19  auto compare = _countUp ?
20  [](const Dim& lhs, const Dim& rhs){ return (lhs < 1) == 0 && lhs <= rhs; } :
21  [](const Dim& lhs, const Dim& rhs){ return (lhs < 1) == 0 && lhs >= rhs; };
22 
23  auto advance = _countUp ?
24  [](Dim& lhs, const Dim& rhs){ lhs += rhs; } :
25  [](Dim& lhs, const Dim& rhs){ lhs -= rhs; } ;
26 
27  for(Dim i = _start; compare(i, _end); advance(i,_step))
28  {
29  // __comment("start : " << this->_start.toString() << ", _end : " << this->_end.toString() << ", _step = " << this->_step.toString() << ", i " << i.toString() << ", comp = " << compare(i,_end));
30 
31  Seconds time = this->prof(i);
32  double op = this->ops(i);
33  double perf = double(op) / double(time.count());
34 
35  this->_elems.push_back(i.prod());
36  this->_times.push_back(time) ;
37  this->_ops.push_back(op) ; // 2 * n^2 + n
38  this->_perf.push_back(perf);
39  }
