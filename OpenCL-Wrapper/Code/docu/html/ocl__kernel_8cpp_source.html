514 std::vector<ocl::Kernel::mem_loc> ocl::Kernel::extractMemlocs(const std::string & kernel)
515 {
516  const auto afterAttributes = kernel.find("void");
517 
518  const size_t start = kernel.find("(",afterAttributes) + 1;
519  const size_t end = kernel.find(")",start) - 1;
520  //TRUE_ASSERT(start < end, "Function not correctly defined.");
521  TRUE_ASSERT((start-1) < (end+1), "Function not correctly defined.");
522 
523  size_t pos_before = start;
524  size_t pos_after = start;
525 
526  vector<mem_loc> locs;
527 
528  while(pos_after <= end)
529  {
530  pos_after = kernel.find(",", pos_before) - 1;
531  if(pos_after > end) pos_after = kernel.find(")", pos_before) - 1;
532  if(pos_after > end) return locs;
533 
534  const string& argument = kernel.substr(pos_before, pos_after - pos_before + 1);
535 
536  if( argument.find("global") != argument.npos || argument.find("__global") != argument.npos )
537  locs.push_back(global);
538  else if( argument.find("local") != argument.npos || argument.find("__local") != argument.npos)
539  locs.push_back(local);
540  else if( argument.find("image") != argument.npos)
541  locs.push_back(image);
542  else if( argument.find("sampler") != argument.npos)
543  locs.push_back(sampler);
544  else
545  locs.push_back(host);
546  pos_before = pos_after+2;
547  }
548  return locs;
549 }

551 std::string ocl::Kernel::extractParameter(const std::string& kernel)
552 {
553  DEBUG_ASSERT(!kernel.empty(), "Function empty");
554  size_t start, end;
555 
556  start = kernel.find("class");
557  if(start == kernel.npos) return "";
558  end = kernel.find(">");
559  TRUE_ASSERT(start < kernel.length(), "Template not correctly defined.");
560  TRUE_ASSERT(end < kernel.length(), "Template not correctly defined.");
561  TRUE_ASSERT(start < end, "Template not correctly defined.");
562  start += 5;
563  end--;
564 
565  const string &substr = kernel.substr(start, end - start + 1);
566 
567  start = substr.find_first_not_of(" ");
568  end = substr.find_last_not_of(" ");
569 
570  //if(start == end) return "";

594 std::string ocl::Kernel::specialize(const std::string& kernel, const std::string& type)//const utl::Type &type)
595 {
596 
597  const string temp_param = ocl::Kernel::extractParameter(kernel);
598 
599  if(temp_param.empty()) return kernel;
600 
601  string fct = kernel;
602 
603  // DEBUG_COMMENT("fct = " << fct);
604 
605 
606  size_t start = 0, end = 0;
607 
608  start = fct.find("template", start);
609  end = fct.find(">",end,1);
610  TRUE_ASSERT(start < end, "Template not correctly defined.");
611 
612  fct.erase(start, end - start + 2);
613 
614 // #if defined OPENCL_V1_1 || defined OPENCL_V1_0
615 //if(type == "double") fct.insert(0, "#if !defined(__OPENCL_VERSION__) || __OPENCL_VERSION__ < 120\n#pragma OPENCL EXTENSION cl_khr_fp64: enable\n#endif\n");
616 // #endif
617 
618 
619  size_t pos = 0;
620 
621  pos = fct.find("void", pos);
622 
623  pos = fct.find("(", pos, 1);
624  TRUE_ASSERT(pos < fct.length(), "Could not find the fct.");
625  fct.insert(pos++,"_",1);
626  //fct.insert(pos,type.name());
627  fct.insert(pos,type);
628  //pos += type.name().length();
629  pos += type.length();
630 
631  const string s1 = temp_param + " ";
632  const string s2 = temp_param + "*";
633  const string s3 = temp_param + "&";
634  const string s4 = temp_param + ")";
635 
636  while(pos < fct.length()){
637  int pos1 = fct.find(s1, pos); if(pos1 == (int)string::npos) pos1 = fct.length();
638  int pos2 = fct.find(s2, pos); if(pos2 == (int)string::npos) pos2 = fct.length();
639  int pos3 = fct.find(s3, pos); if(pos3 == (int)string::npos) pos3 = fct.length();
640  int pos4 = fct.find(s4, pos); if(pos4 == (int)string::npos) pos4 = fct.length();
641 
642  pos = std::min(std::min(std::min(pos1,pos2),pos3),pos4);
643  if(pos >= fct.length()) break;
644  fct.replace(pos, temp_param.length() , type);//type.name());
645 
646  //pos += type.name().length();
647  pos += type.length();
648  }
649 
650  return fct;
651 }
