575 std::string ocl::Program::nextKernel(const std::string &kernels, size_t pos)
576 {
577  if ( pos >= kernels.size() )
578  return "";
579 
580  constexpr char const kernelKeyword1[] = "__kernel";
581  constexpr char const kernelKeyword2[] = "kernel";
582  constexpr char const templateKeyword[] = "template";
583 
584  auto const startTemplate = kernels.find( templateKeyword, pos );
585  auto const startNonTemplate = std::min( kernels.find( kernelKeyword1, pos ), kernels.find( kernelKeyword2, pos ) );
586 
587  size_t end = 0u, start = 0u;
588 
589  if ( startTemplate < startNonTemplate )
590  {
591  start = startTemplate;
592 
593  /*auto const step = std::max( kernels.find( kernelKeyword1, start ), kernels.find( kernelKeyword2, start ) ) + 1u;
594  auto const end1 = kernels.find( templateKeyword, step );
595  auto const end2 = std::min( kernels.find( kernelKeyword1, step ), kernels.find( kernelKeyword2, step ) );
596 
597  end = std::min( end1, end2 );*/
598  }
599  else if ( startTemplate > startNonTemplate )
600  {
601  start = startNonTemplate;
602 
603  /*auto const off = start + sizeof templateKeyword - 1u;
604  auto const end1 = kernels.find( templateKeyword, off );
605  auto const end2 = std::min( kernels.find( kernelKeyword1, off ), kernels.find( kernelKeyword2, off ) );
606 
607  end = std::min( end1, end2 );*/
608  }
609  else // startTemplate == startNonTemplate == std::string::npos
610  {
611  return "";
612  }
613 
614  auto braceFinder = kernels.find( '{', start );
615 
616  if ( braceFinder == std::string::npos )
617  return "";
618 
619  size_t numOpenBraces = 1;
620 
621  while ( numOpenBraces != 0 )
622  {
623  ++braceFinder;
624 
625  if ( braceFinder == kernels.size() )
626  return "";
627 
628  switch ( kernels[braceFinder] )
629  {
630  case '{' : ++numOpenBraces; break;
631  case '}' : --numOpenBraces; break;
632  }
633  }
634 
635  end = braceFinder;
636 
637  if ( _kernels.empty() )
638  commonCodeBlocks_[0] = kernels.substr( pos, start - pos );
639  else
640  commonCodeBlocks_.push_back( kernels.substr( pos, start - pos ) );
641 
642  return kernels.substr( start, end - start +1 );
643 
644 }

652 void ocl::Program::eraseComments(std::string &kernels) const
653 {
654  size_t end_pos = 0, pos = 0;
655  while(pos < kernels.length()){
656  pos = kernels.find("/*", pos,2);
657  end_pos = kernels.find("*/", pos,2);
658  if(pos >= kernels.length()) break;
659  if(end_pos >= kernels.length()) break;
660  TRUE_ASSERT(pos < end_pos, pos << " >= " << end_pos);
661  kernels.erase(pos, end_pos-pos+2);
662  pos += 2;
663  }
664  pos = 0;
665  while(pos < kernels.length()){
666  pos = kernels.find("//", pos,2);
667  end_pos = kernels.find("\n", pos); std::string s("\n");
668  if(pos >= kernels.length()) break;
669  if(end_pos >= kernels.length()) break;
670  TRUE_ASSERT(pos < end_pos, pos << " >= " << end_pos);
671  kernels.erase(pos, end_pos-pos);
672  pos++;

680 void ocl::Program::checkBuild(cl_int buildErr) const
681 {
682  // Exiting the program is not acceptable.
683  if ( buildErr == CL_SUCCESS )
684  return;
685 
686  std::ostringstream oss;
687 
688  bool headerSet = false;
689 
690  for ( auto const& device : _context->devices() )
691  {
692  cl_build_status buildStatus = CL_SUCCESS;
693 
694  clGetProgramBuildInfo( _id, device.id(), CL_PROGRAM_BUILD_STATUS, sizeof buildStatus, &buildStatus, nullptr );
695 
696  if ( buildStatus != CL_SUCCESS )
697  {
698  if ( !headerSet )
699  {
700  oss << "Program failed to build.\n";
701  headerSet = true;
702  }
703 
704  size_t size = 0u;
705 
706  clGetProgramBuildInfo( _id, device.id(), CL_PROGRAM_BUILD_LOG, 0u, nullptr, &size );
707 
708  std::unique_ptr< cl_char[] > buildLog( new cl_char[size] );
709 
710  clGetProgramBuildInfo( _id, device.id(), CL_PROGRAM_BUILD_LOG, size, buildLog.get(), nullptr );
711 
712  oss << "Device " << device.name() << " Build Log:\n" << buildLog.get() << '\n';
713  }
714  }
715 
716  if ( !oss.str().empty() )
717  throw std::runtime_error( oss.str() );
718 }
