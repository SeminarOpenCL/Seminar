112 void ocl::Image::create(size_t width, size_t height, ChannelType type, ChannelOrder order, Access access)
113 {
114  TRUE_ASSERT(this->_context != 0, "Context not valid - cannot create Image");
115  cl_mem_flags flags = access;
116  cl_image_format format;
117  format.image_channel_order = order;
118  format.image_channel_data_type = type;
119  cl_int status;
120 
121 #ifdef OPENCL_V1_2
122  if ( _context->devices()[0].supportsVersion( 1, 2 ) )
123  {
124  cl_image_desc desc;
125  desc.image_type = CL_MEM_OBJECT_IMAGE2D;
126  desc.image_height = height;
127  desc.image_width = width;
128  desc.image_depth = 1;
129  desc.image_array_size = 1;
130  desc.image_row_pitch = 0;
131  desc.image_slice_pitch = 0;
132  desc.num_mip_levels = 0;
133  desc.num_samples = 0;
134  desc.buffer = NULL;
135 
136  this->_id = clCreateImage(this->_context->id(), flags, &format, &desc, NULL, &status);
137  }
138  else
139 #endif
140  {
141  this->_id = clCreateImage2D(this->_context->id(), flags, &format, width, height, 0, NULL, &status);
142  }
143 
144  OPENCL_SAFE_CALL(status);
145  TRUE_ASSERT(this->_id != 0, "Could not create 2D image.");
146 }

206 void ocl::Image::create(unsigned int texture, unsigned long texture_target, long miplevel)
207 {
208  cl_mem_flags flags = ocl::Image::ReadWrite;
209  if (this->_context->devices().size() == 1 &&
210  this->_context->devices().front().type() == ocl::device_type::CPU){
211  flags |= ocl::Image::AllocHost;
212  }
213 
214  cl_int status;
215 #ifdef OPENCL_V1_2
216  this->_id = clCreateFromGLTexture(this->_context->id(), flags, texture_target, miplevel, texture, &status);
217 #else
218  this->_id = clCreateFromGLTexture2D(this->_context->id(), flags, texture_target, miplevel, texture, &status);
219 #endif
220  OPENCL_SAFE_CALL(status);
221  TRUE_ASSERT(this->_id != 0, "Could not create shared image.");
222 }
